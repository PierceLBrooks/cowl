# Directories

set(COWL_SERIALIZER_SRC_DIR "${CMAKE_CURRENT_SOURCE_DIR}")
list(APPEND COWL_PRIVATE_HEADERS_DIRS "${COWL_SERIALIZER_SRC_DIR}")

# Sources

file(GLOB COWL_SERIALIZER_SOURCES CONFIGURE_DEPENDS
     "${COWL_SERIALIZER_SRC_DIR}/*.c")
list(APPEND COWL_SOURCES ${COWL_SERIALIZER_SOURCES})

# Propagate changes in the parent project

set(COWL_PRIVATE_HEADERS_DIRS ${COWL_PRIVATE_HEADERS_DIRS} PARENT_SCOPE)
set(COWL_SOURCES ${COWL_SOURCES} PARENT_SCOPE)
